<script>
// SETTINGS
const SHOW_THRESHOLD=100000;
const REBIRTH_REQUIREMENT=5000000;
let currentUser=null, data=null, auto=false;

// EXPANDED RARITIES
const rarities=[
{name:"Moss",odds:120000},{name:"Vine",odds:150000},{name:"Bloom",odds:200000},{name:"Oakheart",odds:300000},
{name:"Starflower",odds:500000},{name:"Crystal Fern",odds:750000},
{name:"Dust",odds:400000},{name:"Dune",odds:600000},{name:"Sandstorm",odds:800000},{name:"Desert King",odds:1200000},{name:"Mirage",odds:1500000},{name:"Golden Hourglass",odds:2000000},
{name:"Magma",odds:2000000},{name:"Tempest",odds:5000000},{name:"Tectonic",odds:10000000},{name:"Primordial",odds:25000000},{name:"Void",odds:40000000},{name:"Quantum Flame",odds:60000000},
{name:"Nebula",odds:50000000},{name:"Singularity",odds:100000000},{name:"Multiverse",odds:250000000},{name:"Eternity Core",odds:500000000},{name:"Absolute GOD",odds:1000000000},{name:"Ultra GOD",odds:2000000000}
];

// ACCOUNT SYSTEM
function createAccount(){
 let u=username.value, p=password.value;
 if(!u||!p){loginMsg.innerText="Fill all fields";return;}
 let accounts=JSON.parse(localStorage.getItem("accounts"))||{};
 if(accounts[u]){loginMsg.innerText="Username exists";return;}
 accounts[u]={password:p,gameData:newGameData()};
 localStorage.setItem("accounts",JSON.stringify(accounts));
 loginMsg.innerText="Account created!";
}
function login(){
 let u=username.value, p=password.value;
 let accounts=JSON.parse(localStorage.getItem("accounts"))||{};
 if(!accounts[u]||accounts[u].password!==p){loginMsg.innerText="Invalid login";return;}
 currentUser=u; data=accounts[u].gameData; startGame();
}
function logout(){auto=false; saveGame(); location.reload();}
function startGame(){loginScreen.style.display="none"; game.style.display="flex"; document.getElementById("currentUser").innerText=currentUser; checkUnlocks(); updateUI();}
function newGameData(){return{rolls:0,rebirths:0,luck:1,inventory:{},top10:[],sessionTop:[]};}

// ROLL SYSTEM
function rollBulk(amount){
 data.rolls+=amount;
 rarities.forEach(r=>{
  let adjusted=r.odds/data.luck;
  let chance=1-Math.pow((1-1/adjusted),amount);
  if(Math.random()<chance){registerRoll(r);}
 });
 checkUnlocks(); updateUI(); saveGame();
}
function registerRoll(r){
 document.getElementById("lastRoll").innerHTML=`<span class="glow">${r.name} (1/${r.odds.toLocaleString()})</span>`;
 data.inventory[r.name]=(data.inventory[r.name]||0)+1;
 if(r.odds>=SHOW_THRESHOLD){addRecent(r); addTop(r,data.top10,"leaderboard"); addTop(r,data.sessionTop,"sessionTop");}
}
function addRecent(r){let div=document.createElement("div"); div.className="box glow"; div.innerText=r.name+" (1/"+r.odds.toLocaleString()+")"; recent.prepend(div); if(recent.children.length>50){recent.removeChild(recent.lastChild);}}
function addTop(r,array,id){array.push(r); array.sort((a,b)=>b.odds-a.odds); array.splice(10); const board=document.getElementById(id); board.innerHTML=""; array.forEach((item,i)=>{let div=document.createElement("div"); div.className="box glow"; div.innerText=(i+1)+". "+item.name+" (1/"+item.odds.toLocaleString()+")"; board.appendChild(div);});}

// REBIRTH
function rebirth(){if(data.rolls<REBIRTH_REQUIREMENT){alert("Need 5M rolls!");return;} data.rebirths++; data.luck=1+data.rebirths*0.25; data.rolls=0; data.inventory={}; data.sessionTop=[]; document.getElementById("recent").innerHTML=""; alert("Rebirth complete! Luck boosted."); updateUI(); saveGame();}

// AUTO-ROLL
function toggleAuto(){auto=!auto; if(auto){autoLoop();}}
function autoLoop(){if(!auto) return; rollBulk(getHighestUnlocked()); setTimeout(autoLoop,200);}
function getHighestUnlocked(){if(data.rolls>=1000000) return 10000000; if(data.rolls>=100000) return 1000000; if(data.rolls>=10000) return 100000; if(data.rolls>=100) return 100; return 1;}

// UNLOCKS
function checkUnlocks(){unlockIf("btn10",100,"x10"); unlockIf("btn100",10000,"x100"); unlockIf("btn1m",100000,"x1M"); unlockIf("btn10m",1000000,"x10M"); unlockIf("btn100m",10000000,"x100M");}
function unlockIf(id,req,label){if(data.rolls>=req) unlock(id,label);}
function unlock(id,label){let btn=document.getElementById(id); if(btn.disabled){btn.disabled=false; btn.innerText=label;}}

// UI
function updateUI(){
 rollCount.innerText=data.rolls.toLocaleString();
 rebirthCount.innerText=data.rebirths;
 luckDisplay.innerText=data.luck.toFixed(2)+"x";
 inventory.innerHTML="";
 for(let key in data.inventory){let div=document.createElement("div"); div.className="box"; div.innerText=key+" x"+data.inventory[key]; inventory.appendChild(div);}
}

// SAVE
function saveGame(){let accounts=JSON.parse(localStorage.getItem("accounts"))||{}; accounts[currentUser].gameData=data; localStorage.setItem("accounts",JSON.stringify(accounts));}
